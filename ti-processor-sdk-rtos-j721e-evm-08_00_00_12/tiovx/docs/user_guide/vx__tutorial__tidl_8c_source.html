163  vx_context context;
164  vx_user_data_object config1, config2, realConfig;
165  vx_user_data_object network;
166  vx_user_data_object createParams1[MAX_NUM_THREADS];
167  vx_user_data_object inArgs1[MAX_NUM_THREADS];
168  vx_user_data_object outArgs1[MAX_NUM_THREADS];
169  vx_user_data_object createParams2[MAX_NUM_THREADS];
170  vx_user_data_object inArgs2[MAX_NUM_THREADS];
171  vx_user_data_object outArgs2[MAX_NUM_THREADS];
172  vx_reference params[5];
173  vx_tensor input_tensors[MAX_NUM_THREADS][VX_TUTORIAL_MAX_TENSORS];
174  vx_tensor output_tensors1[MAX_NUM_THREADS][VX_TUTORIAL_MAX_TENSORS];

180  uint64_t exe_time=0;

182  size_t sizeFilePath;
183  char filePath[TIOVX_UTILS_MAXPATHLENGTH];

188  vx_enum targetCpuId2[MAX_NUM_THREADS];
189 
190  vx_status status = (vx_status)VX_SUCCESS;
191 
192  VxTutorialTidl_CfgObj *obj = &gCfgObj;
193 
194  vx_graph graph[MAX_NUM_THREADS] = {0};
195  vx_node node1[MAX_NUM_THREADS] = {0};
196  vx_node node2[MAX_NUM_THREADS] = {0};
197  vx_kernel kernel1 = 0;
198  vx_kernel kernel2 = 0;
199 
200  uint32_t num_input_tensors = 0;
201  uint32_t num_output_tensors1 = 0;
202  uint32_t num_output_tensors2 = 0;
203 
204  uint32_t maxNumThreads= 1;

206  printf(" vx_tutorial_tidl: Tutorial Started !!! \n");
207 
208  context = vxCreateContext();

211  vxDirective((vx_reference)context, (vx_enum)VX_DIRECTIVE_ENABLE_PERFORMANCE);

213  basePath = tivx_utils_get_test_file_dir();
214

217  goto exit;

220  sizeFilePath = snprintf(filePath, TIOVX_UTILS_MAXPATHLENGTH, "%s/%s", basePath, CFG_FILE_NAME);
221 
222  if (sizeFilePath > TIOVX_UTILS_MAXPATHLENGTH) {
223  printf("Error: path of config gile too long to fit in string\n");
224  goto exit;

227  printf(" Reading config file %s ...\n", filePath);
228 
229  status= parse_cfg_file(obj, filePath);
230  if (status!=(vx_status)VX_SUCCESS) {
231  goto exit;
232  }
233 
234 
235  printf(" Reading network file %s ...\n", obj->tidl_network_file_path);
236 
237  network = vx_tidl_utils_readNetwork(context, &obj->tidl_network_file_path[0]);
238  VX_TUTORIAL_ASSERT_VALID_REF(network)
239 
240  /*
241  * Processing_core_mode: Specify how the network will be processed if multiple processing cores exist in the system.
242  * 0 (default): all cores can be utilized according to each layer's groupID. If a layer's group ID is 1 then it will run on EVE1. If it is 2, it will run on DSP1.
243  * 1: The entire network will run on EVE1, even the layers which have group ID 2 (DSP layers).
244  * 2: The entire network will run on DSP1, even the layers which have group ID 1 (EVE layers).
245  *
246  */
247 #ifdef HOST_EMULATION
248  /* In host emulation on PC, it is not possible to test for processing_core_mode=2
249  * but we can test test processing_core_mode=1, which gives a high confidence that
250  * processing_core_mode=2 works as well.
251  * For a definitive testing of processing_core_mode=2, test on target.
252  */
253  if (obj->processing_core_mode== 2) {
254  obj->processing_core_mode= 1;
255  }
256 #endif
257 
258  if (obj->processing_core_mode== 0){
259 
260  /*
261  * In case the network has only one group of layer, assign the first core and disable the second core
262  * In case there are 2 groups of layers. 1st group is always assigned to EVE and second group always assigned to DSP
263  * */
264  int32_t layersGroupCount[(vx_enum)TIVX_CPU_ID_MAX];
265  int32_t numLayersGroup= vx_tidl_utils_countLayersGroup(network, layersGroupCount);
266 
267  if (numLayersGroup== 1) {
268  if (layersGroupCount[1]!=0) {
269  /* If the entire network runs on EVE, spun the processing into as many threads as there are EVEs to demonstrate parallelism between EVEs */
270  maxNumThreads= NUM_EVE_CPU;
271  targetCore1[0]= TIVX_TARGET_EVE1;targetCore1[1]= TIVX_TARGET_EVE2;targetCore1[2]= TIVX_TARGET_EVE3;targetCore1[3]= TIVX_TARGET_EVE4;
272  targetCpuId1[0]= (vx_enum)TIVX_CPU_ID_EVE1;targetCpuId1[1]= (vx_enum)TIVX_CPU_ID_EVE2;targetCpuId1[2]= (vx_enum)TIVX_CPU_ID_EVE3;targetCpuId1[3]= (vx_enum)TIVX_CPU_ID_EVE4;
273  }
274  else if (layersGroupCount[2]!=0) {
275  /* If the entire network runs on DSP, spun the processing into as many threads as there are DSPs to demonstrate parallelism between DSPs */
276  maxNumThreads= NUM_DSP_CPU;
277  targetCore1[0]= TIVX_TARGET_DSP1;targetCore1[1]= TIVX_TARGET_DSP2;
278  targetCpuId1[0]= (vx_enum)TIVX_CPU_ID_DSP1;targetCpuId1[1]= (vx_enum)TIVX_CPU_ID_DSP2;
279  }
280  else {
281  printf(" Invalid layer group ID detected, exiting ...\n");
282  goto exit;
283  }
284  for (threadIdx= 0; threadIdx < maxNumThreads; threadIdx++) {
285  targetCore2[threadIdx]= NULL;
286  targetCpuId2[threadIdx]= (vx_enum)TIVX_CPU_ID_INVALID;
287  }
288  }/* In case the network has 2 groups, it will run on two cores */
289  else if (numLayersGroup== 2) {
290  maxNumThreads= NUM_EVE_CPU;
291  targetCore1[0]= TIVX_TARGET_EVE1;targetCore1[1]= TIVX_TARGET_EVE2;targetCore1[2]= TIVX_TARGET_EVE3;targetCore1[3]= TIVX_TARGET_EVE4;
292  targetCpuId1[0]= (vx_enum)TIVX_CPU_ID_EVE1;targetCpuId1[1]= (vx_enum)TIVX_CPU_ID_EVE2;targetCpuId1[2]= (vx_enum)TIVX_CPU_ID_EVE3;targetCpuId1[3]= (vx_enum)TIVX_CPU_ID_EVE4;
293  targetCore2[0]= TIVX_TARGET_DSP1;targetCore2[1]= TIVX_TARGET_DSP1;targetCore2[2]= TIVX_TARGET_DSP1;targetCore2[3]= TIVX_TARGET_DSP1;
294  targetCpuId2[0]= (vx_enum)TIVX_CPU_ID_DSP1;targetCpuId2[1]= (vx_enum)TIVX_CPU_ID_DSP1;targetCpuId2[2]= (vx_enum)TIVX_CPU_ID_DSP1;targetCpuId2[3]= (vx_enum)TIVX_CPU_ID_DSP1;
295  }
296  else {
297  printf(" Invalid number of groups of layers, exiting ...\n");
298  goto exit;
299  }
300 
301  }
302  else if (obj->processing_core_mode== 1) {
303  maxNumThreads= NUM_EVE_CPU;
304  targetCore1[0]= TIVX_TARGET_EVE1;targetCore1[1]= TIVX_TARGET_EVE2;targetCore1[2]= TIVX_TARGET_EVE3;targetCore1[3]= TIVX_TARGET_EVE4;
305  targetCpuId1[0]= (vx_enum)TIVX_CPU_ID_EVE1;targetCpuId1[1]= (vx_enum)TIVX_CPU_ID_EVE2;targetCpuId1[2]= (vx_enum)TIVX_CPU_ID_EVE3;targetCpuId1[3]= (vx_enum)TIVX_CPU_ID_EVE4;
306  for (threadIdx= 0; threadIdx < maxNumThreads; threadIdx++) {
307  targetCore2[threadIdx]= NULL;
308  targetCpuId2[threadIdx]= (vx_enum)TIVX_CPU_ID_INVALID;
309  }
310  }
311  else if (obj->processing_core_mode== 2) {
312  maxNumThreads= NUM_DSP_CPU;
313  targetCore1[0]= TIVX_TARGET_DSP1;targetCore1[1]= TIVX_TARGET_DSP2;
314  targetCpuId1[0]= (vx_enum)TIVX_CPU_ID_DSP1;targetCpuId1[1]= (vx_enum)TIVX_CPU_ID_DSP2;
315  for (threadIdx= 0; threadIdx < maxNumThreads; threadIdx++) {
316  targetCore2[threadIdx]= NULL;
317  targetCpuId2[threadIdx]= (vx_enum)TIVX_CPU_ID_INVALID;
318  }
319  }
320  else {
321  printf("Invalid processing core mode, exiting ...\n");
322  goto exit;
323  }
324 
325  /* If processing_core_mode is not 0, update each layer's group ID so that the entire network runs either on EVE or DSP*/
326  if (obj->processing_core_mode!= 0) {
327  vx_tidl_utils_updateLayersGroup(network, targetCpuId1[0]);
328  }
329 
330  config1 = vx_tidl_utils_getConfig(context, network, &num_input_tensors, &num_output_tensors1, targetCpuId1[0]);
331 
332  /* In case the network runs on one CPU, set num_output_tensors2 to 0 */
333  if (targetCpuId2[0]== (vx_enum)TIVX_CPU_ID_INVALID) {
334  num_output_tensors2= 0;
335  config2= 0;
336  }
337  else {
338  int32_t num_interm_tensors= num_output_tensors1;
339 
340  config2 = vx_tidl_utils_getConfig(context, network, &num_output_tensors1, &num_output_tensors2, targetCpuId2[0]);
341 
342  if (num_interm_tensors != num_output_tensors1) {
343  printf("Number of output tensors from first group of layers not equal to the number of input tensors from second group of layers. Exiting ...\n");
344  goto exit;
345  }
346  }
347 
348  printf(" Reading network params file %s ...\n", obj->tidl_params_file_path);
349 
350  status= vx_tidl_utils_readParams(network, &obj->tidl_params_file_path[0]);
351  VX_TUTORIAL_ASSERT(status==(vx_status)VX_SUCCESS);
352 
353  kernel1 = tivxAddKernelTIDL(context, num_input_tensors, num_output_tensors1);
354  VX_TUTORIAL_ASSERT_VALID_REF(kernel1)
355 
356  if (targetCpuId2[0]!= (vx_enum)TIVX_CPU_ID_INVALID) {
357  kernel2 = tivxAddKernelTIDL(context, num_output_tensors1, num_output_tensors2);
358  VX_TUTORIAL_ASSERT_VALID_REF(kernel2)
359  }
360 
361 
362  for (threadIdx= 0; threadIdx < maxNumThreads; threadIdx++) {
363 
364  printf("\nThread #%d: Create graph ... \n", threadIdx+1);
365 
366  /* Create OpenVx Graph */
367  graph[threadIdx] = vxCreateGraph(context);
368  VX_TUTORIAL_ASSERT_VALID_REF(graph[threadIdx])
369 
370  printf("Thread #%d: Create input and output tensors for node 1 ... \n", threadIdx+1);
371  /* Create array of input tensors for the first node */
372 
373  status= createInputTensors(context, config1, &input_tensors[threadIdx][0]);
374  VX_TUTORIAL_ASSERT(status==(vx_status)VX_SUCCESS);
375 
376  /* Create array of output tensors for the first node, which is also the input tensors for the second node */
377  status= createOutputTensor(context, config1, &output_tensors1[threadIdx][0]);
378  VX_TUTORIAL_ASSERT(status==(vx_status)VX_SUCCESS);
379 
380  /*
381  * TIDL maintains range statistics for previously processed frames. It quantizes the current inference activations using range statistics from history for processes (weighted average range).
382  * Below is the parameters controls quantization.
383  * quantMargin is margin added to the average in percentage.
384  * quantHistoryBoot weights used for previously processed inference during application boot time for initial few frames
385  * quantHistory weights used for previously processed inference during application execution (After initial few frames)
386  *
387  * Below settings are adequate for running on videos sequences.
388  * For still images, set all settings to 0.
389  */
390  quantHistoryBoot= 20;
391  quantHistory= 5;
392  quantMargin= 0;
393 
394  printf("Thread #%d: Create node 1 ... \n", threadIdx+1);
395 
396  createParams1[threadIdx]= vx_tidl_utils_setCreateParams(context, quantHistoryBoot, quantHistory, quantMargin);
397  inArgs1[threadIdx]= vx_tidl_utils_setInArgs(context);
398  outArgs1[threadIdx]= vx_tidl_utils_setOutArgs(context);
399 
400  params[0]= (vx_reference)config1;
401  params[1]= (vx_reference)network;
402  params[2]= (vx_reference)createParams1[threadIdx];
403  params[3]= (vx_reference)inArgs1[threadIdx];
404  params[4]= (vx_reference)outArgs1[threadIdx];
405 
406  node1[threadIdx] = tivxTIDLNode(graph[threadIdx], kernel1,
407  params,
408  &input_tensors[threadIdx][0],
409  &output_tensors1[threadIdx][0]
410  );
411  VX_TUTORIAL_ASSERT_VALID_REF(node1[threadIdx])
412 
413  /* Set target node to targetCore1 (EVEn or DSP1)*/
414  vxSetNodeTarget(node1[threadIdx], (vx_enum)VX_TARGET_STRING, targetCore1[threadIdx]);
415 #ifdef HOST_EMULATION
416  /* This is a workaround to support spanning graphs on different EVE and DSP cores in PC host emulation environment
417  * */
418  gTidlNodeCpuId[2*threadIdx]= targetCpuId1[threadIdx];
419 #endif
420 
421  if ((targetCpuId2[threadIdx]== (vx_enum)TIVX_CPU_ID_DSP1) || (targetCpuId2[threadIdx]== (vx_enum)TIVX_CPU_ID_DSP2)) {
422  printf("Thread #%d: Create output tensors for node 2 ... \n", threadIdx+1);
423 
424  /* Create array of output tensors for the second node */
425  status= createOutputTensor(context, config2, &output_tensors2[threadIdx][0]);
426  VX_TUTORIAL_ASSERT(status==(vx_status)VX_SUCCESS);
427 
428  printf("Thread #%d: Create node 2 ... \n", threadIdx+1);
429 
430  createParams2[threadIdx]= vx_tidl_utils_setCreateParams(context, quantHistoryBoot, quantHistory, quantMargin);
431  inArgs2[threadIdx]= vx_tidl_utils_setInArgs(context);
432  outArgs2[threadIdx]= vx_tidl_utils_setOutArgs(context);
433 
434  params[0]= (vx_reference)config2;
435  params[1]= (vx_reference)network;
436  params[2]= (vx_reference)createParams2[threadIdx];
437  params[3]= (vx_reference)inArgs2[threadIdx];
438  params[4]= (vx_reference)outArgs2[threadIdx];
439 
440  node2[threadIdx] = tivxTIDLNode(graph[threadIdx], kernel2,
441  params,
442  &output_tensors1[threadIdx][0],
443  &output_tensors2[threadIdx][0]
444  );
445  VX_TUTORIAL_ASSERT_VALID_REF(node2[threadIdx])
446 
447  /* Set target node to targetCore2 (EVEn or DSP1)*/
448  vxSetNodeTarget(node2[threadIdx], (vx_enum)VX_TARGET_STRING, targetCore2[threadIdx]);
449 #ifdef HOST_EMULATION
450  /* This is a workaround to support spanning graphs on different EVE and DSP cores in PC host emulation environment
451  * */
452  gTidlNodeCpuId[2*threadIdx+1]= targetCpuId2[threadIdx];
453 #endif
454 
455  }
456  }
457 
458  printf("\n");
459 
460  for (threadIdx= 0; threadIdx < maxNumThreads; threadIdx++) {
461  printf("Thread #%d: Verify graph ... \n", threadIdx+1);
462  /* Verify the TIDL Graph
463  * When executed in host emulation on PC, the version of TI-DL library linked displays information about each layer of the network.
464  * In target execution, such display is disabled in the library.
465  * */
466  status = vxVerifyGraph(graph[threadIdx]);
467  VX_TUTORIAL_ASSERT(status==(vx_status)VX_SUCCESS);
468  }
469 
470  if((vx_status)VX_SUCCESS == status) {
471 
472  for (threadIdx= 0; threadIdx < maxNumThreads; threadIdx++) {
473 
474  /* Read input from file and populate the input tensor #0, we assume here that only one input tensor is used */
475  status= readInput(context, config1, &input_tensors[threadIdx][0], &obj->input_file_path[0], obj->operation_mode);
476  VX_TUTORIAL_ASSERT(status==(vx_status)VX_SUCCESS);
477 
478  if (status!=(vx_status)VX_SUCCESS) {
479  goto exit;
480  }
481 
482  }
483 
484  exe_time= tivxPlatformGetTimeInUsecs();
485 
486  printf("\n");
487 
488 #ifdef SEQUENTIAL_SCHEDULE
489  for (threadIdx= 0; threadIdx < maxNumThreads; threadIdx++) {
490  printf("Thread #%d: Execute graph ... \n",threadIdx + 1);
491  /* Execute the network */
492  status = vxProcessGraph(graph[threadIdx]);
493  VX_TUTORIAL_ASSERT(status==(vx_status)VX_SUCCESS);
494  }
495 #else
496  for (threadIdx= 0; threadIdx < maxNumThreads; threadIdx++) {
497  printf("Thread #%d: Start graph ... \n",threadIdx + 1);
498  /* Execute the network */
499  status = vxScheduleGraph(graph[threadIdx]);
500  VX_TUTORIAL_ASSERT(status==(vx_status)VX_SUCCESS);
501  }
502 
503  /* You can do other useful things here, while the graphs execute asynchronously using resources available on other cores */
504  printf("\n");
505 
506  for (threadIdx= 0; threadIdx < maxNumThreads; threadIdx++) {
507  printf("Thread #%d: Wait for graph ... \n",threadIdx + 1);
508  /* Execute the network */
509  status = vxWaitGraph(graph[threadIdx]);
510  VX_TUTORIAL_ASSERT(status==(vx_status)VX_SUCCESS);
511  }
512 #endif
513 
514  exe_time= tivxPlatformGetTimeInUsecs() - exe_time;
515 
516  for (threadIdx= 0; threadIdx < maxNumThreads; threadIdx++) {
517 
518  printf("\nThread #%d: Results\n", threadIdx+1);
519  printf("---------------------\n");
520 
521  /* Display the output_tensors1 if graph runs 1 cores */
522  if (targetCpuId2[threadIdx]== (vx_enum)TIVX_CPU_ID_INVALID) {
523  real_output_tensors= &output_tensors1[threadIdx][0];
524  realConfig= config1;
525  }
526  else { /* Display the output_tensors2 if graph runs 2 cores */
527  real_output_tensors= &output_tensors2[threadIdx][0];
528  realConfig= config2;
529  }
530 
531  displayOutput(NULL, (vx_df_image)NULL, NULL, 0, 0, 0, realConfig, real_output_tensors, &obj->output_file_path[0], obj->operation_mode);
532 
533  }
534  }
535
