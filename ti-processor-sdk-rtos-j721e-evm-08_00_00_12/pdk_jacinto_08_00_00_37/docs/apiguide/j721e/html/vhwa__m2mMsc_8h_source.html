336 static inline void Vhwa_M2mMscSl2AllocPrmsInit(Vhwa_M2mMscSl2AllocPrms *sl2Prms)

339 
340  if (NULL != sl2Prms)

342  for(cnt = 0; cnt < VHWA_M2M_MSC_MAX_INPUT_BUFF; cnt++)
343  {
344  sl2Prms->maxInWidth[cnt] = 1920U;
345  sl2Prms->inCcsf[cnt] = FVID2_CCSF_BITS12_UNPACKED16;
346  sl2Prms->inBuffDepth[cnt] = 6U;
347  }
348 
349  for(cnt = 0; cnt < MSC_MAX_OUTPUT; cnt++)
350  {
351  sl2Prms->maxOutWidth[cnt] = 1920U;
352  sl2Prms->outCcsf[cnt] = FVID2_CCSF_BITS12_UNPACKED16;
353  sl2Prms->outBuffDepth[cnt] = 2U;
354  }
355  }
