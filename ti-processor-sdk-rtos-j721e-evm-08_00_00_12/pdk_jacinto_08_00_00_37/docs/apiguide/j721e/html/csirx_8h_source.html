556 static inline void Csirx_createParamsInit(Csirx_CreateParams *createPrms)
557 {
558  uint32_t loopCnt;
559 
560  if (NULL != createPrms)
561  {
562  createPrms->numCh = 0U;
563  for (loopCnt = 0U ; loopCnt < CSIRX_NUM_CH ; loopCnt++)
564  {
565  Csirx_chCfgInit(&createPrms->chCfg[loopCnt]);
566  }
567  Csirx_instCfgInit(&createPrms->instCfg);
568  createPrms->frameDropBufLen = 0U;
569  createPrms->frameDropBuf = 0U;
570  createPrms->enableChDq = 0U;
571  }
572 }

574 static inline void Csirx_chCfgInit(Csirx_ChCfg *chCfgPrms)
575 {
576  chCfgPrms->chId = CSIRX_CH_ID_DEFAULT;
577  chCfgPrms->chType = CSIRX_CH_TYPE_CAPT;
578  chCfgPrms->vcNum = CSIRX_VC_NUM_DEFAULT;
579  chCfgPrms->inCsiDataType = CSIRX_IN_DT_DEFAULT;
580  Fvid2Format_init(&chCfgPrms->outFmt);
581  chCfgPrms->outFmt.width = 0U;
582  chCfgPrms->outFmt.height = 0U;
583  Fvid2Utils_memset(&chCfgPrms->outFmt.pitch[0U],
584  0x0,
585  sizeof (chCfgPrms->outFmt.pitch));
586 
587  /* Initialization of UDMA channel params */
588  chCfgPrms->ringOrderId = UDMA_DEFAULT_RING_ORDER_ID;
589  UdmaChRxPrms_init(&chCfgPrms->rxChParams, UDMA_CH_TYPE_RX);
590  chCfgPrms->rxChParams.chanType = CSL_UDMAP_CHAN_TYPE_REF_TR_RING;
591  /* Make CSI2RX DMA channels as highest priority,
592  it is assumed that order 8-15 is mapped as RT at system level */
593  chCfgPrms->rxChParams.busPriority = 0U;
594  chCfgPrms->rxChParams.busQos = 0U;
595  chCfgPrms->rxChParams.busOrderId = 8U;
596  chCfgPrms->rxChParams.dmaPriority =
597  TISCI_MSG_VALUE_RM_UDMAP_CH_SCHED_PRIOR_HIGH;
598 }
